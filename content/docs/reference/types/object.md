# `object`

```ts
type ObjectField = {
  label: string
  name: string
  type: 'object'
  /** `fields OR `templates` may be provided, not both **/
  fields?: Field[]
  templates?: Template[]
  /** See docs/reference/toolkit/fields for customizing the UI **/
  ui?: object
}
```
